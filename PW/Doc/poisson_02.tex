A slight modification for {\tt Poisson\_solve}

\begin{juliacode}
function Poisson_solve( PW::PWGrid, rhoR )
  G2 = PW.gvec.G2
  Ns = PW.Ns
  Npoints = prod(Ns)
  ctmp = 4.0*pi*R_to_G( Ns, rhoR )
  ctmp[1] = 0.0
  for ip = 2:Npoints
    ctmp[ip] = ctmp[ip]/G2[ip]
  end
  return ctmp
end
\end{juliacode}
